jQuery学习（二） 常用选择器

选择器手册

1. 常用基础选择器
  1. 元素选择器

$("标签名")

var p = $('p');
  1. #id 选择器

$("#id名")

var idName = $('#my-id');
  1. .class选择器

$(".class名")

var className = $('#my-class');

4 当前对象选择器

$(this)

var thisObj = $(this);   //选取当前 HTML 元素
2. 复合选择器

注意：jQuery的选择器大部分都是基于css已经存在的选择器

  1. 交集选择器

$("标签选择器[Id选择器|类选择器]")

var pNav = $('p.nav');
  1. 并集选择器

$("选择器,选择器,选择器...");

var navs = $("div,.bg,#nav,p");
  1. 子元素选择器

$("选择器 > 选择器");

var a = $("div > a");
  1. 后代选择器

$("选择器 选择器");

var allA = $("div a");
  1. 兄弟相邻选择器

$("选择器+选择器");

var div = $("div+p");

说明：
1.根据定义的规则查找对象
2.查找到对象是规则的后者

  1. 属性选择器

$(“选择器[属性名称=‘值’]”);

var hrefUrl = $("a[href='http://www.baidu.com']");
3. 伪类选择器
  1. :checked
  2. :selected
<input type="radio" name='sex' value="男" checked />
<input type="radio" name='sex' value="女"/>
<select name="xq" id="">
		<option value="1" selected>1</option>
		<option value="2">2</option>
	</select>
<script type="text/javascript" src="js/jquery-3.3.1.min.js"></script>
<script type="text/javascript">
$(function(){
	var sexObj = $("input[name='sex']:checked");
	console.log(sexObj.length);
	console.log(sexObj.val());	//获取到选中的value
	console.log(sexObj.value);	//undefined sexObj[0].value能获取到值
	var selected = $("select[name='xq']:selected");
})
</script>
4.伪元素选择器
  1. nth-of-type(n|odd|even)

$("选择器:nth-of-type(n|odd|even)")

var nth = $("li:nth-of-type(1)");
  1. nth-child(n|odd|even)

$("选择器:nth-child(n|odd|even)")

var child = $("li:nth-child(1)");
  1. first-child

$("选择器:first-child")

var p = $("p:first-child")

4.last-child

$("选择器:last-child")
