四、数组合并

concat() 方法用于连接两个或多个数组。
语法：
数组1.concat(数组2)
数组1.concat(数组2,数组3)

var arr1 = ["a","b","c"];
var arr2 = ["a","b","c"];
var arr3 = ["g","h","k"];
var arr4 = arr1.concat(arr3);//输出： a,b,c,g,h,k 
var arr4 = arr1.concat(arr2,arr3);//输出： a,b,c,a,b,c,g,h,k 
document.write(arr4)
五、数组排序

sort() 方法用于对数组的元素进行排序

//1.传入一个方法带参数 a-b排序
var arr = [2,8,12,10,9,6];
var arr2 = arr.sort(function(a,b){
   return a-b;
});
document.write(arr2);//输出： 2,6,8,9,10,12 
//2.传入一个方法带参数 b-a排序
var arr2 = arr.sort(function(a,b){
   return b-a;
});
document.write(arr2);//输出： 12,10,9,8,6,2 
//3.可以做用户列表排序
var arr = new Array(6)
arr[0] = "George"

冒泡排序 理解冒泡排序是锻炼逻辑思维的一种方式,不急于立即理解。 后期多看就可以加深对数组和循环的了解

<script type="text/javascript">
    /* [1,2,3,4,5,6,7,8,9] */
    // 需要排序的数组
    var arr = [1, 5, 3, 2, 4, 7, 6, 8, 9, 88, 75, 64, 0];
    // 外层数据循数据的个数就是需要循环的次数,用来执行 length次循环,调出所有数据去进行大小比对
    for (var i = 0; i < arr.length; i++) {
        // 取出数据进行大小比对,执行一次相邻数值大的向前移动一位。执行 length次所有的数据都对比完
        for (var k = 0; k < arr.length; k++) {
            // 相邻的两个数据进行大小比对
            if (arr[k] > arr[k + 1]) {
                // 创建一个存储变量存储比较大的数值
                temp = arr[k];
                // 把小的值索引减一,也就是把小的数字换到前面
                arr[k] = arr[k + 1];
                // 然后把大的数值放在下一位
                arr[k + 1] = temp;
            }
        }
    }
    console.log(arr);
</script>
