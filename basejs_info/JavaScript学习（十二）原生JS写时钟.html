	//页面加载
    window.onload = function () {
        //处理如果时分秒小于10不能显示个位数要显示两位数方法处理 比如：01的格式
        function toDou(n) {
            if (n < 10) {
                return "0" + n;
            } else {
                return "" + n;
            }
        }

        // 时间变化方法定义
        function times() {
            // 创建对象,获取时分秒
            var mydate = new Date();
            var h = mydate.getHours();
            var m = mydate.getMinutes();
            var s = mydate.getSeconds();
            // 时间栏里面的时间创建,利用DOM id="time"
            var time = document.getElementById("time");
            // 获取time下面的三个span标签
            var span = time.getElementsByTagName("span");
            // 第一个标签为小时,使用上面的toDou方法变化一下单数时间格式
            span[0].innerHTML = toDou(h);
            // 第二个标签为分钟
            span[1].innerHTML = toDou(m);
            // 第三个标签为秒数
            span[2].innerHTML = toDou(s);
            // DOM元素时分秒
            var hour = document.getElementById("hours");
            var minute = document.getElementById("minutes");
            var second = document.getElementById("seconds");
            // 旋转时分秒指针
            hour.style.transform = "rotate(" + (h * 30 - 90 + 0.5 * m) + "deg)";
            minute.style.transform = "rotate(" + ((m * 6) - 90) + "deg)";
            second.style.transform = "rotate(" + ((s * 6) - 90) + "deg)";
        }

        // 计时器显示时间数据
        setInterval(times, 1000);
        times();
    }
</script>

<!DOCTYPE html>

            position: relative;
        }

<body>
<div class="container">
    <span id="back"></span>
    <div id="box">
        <ul>
            <li><img src="images/1.jpg" alt=""></li>
            <li><img src="images/2.jpg" alt=""></li>
            <li><img src="images/3.jpg" alt=""></li>
            <li><img src="images/4.jpg" alt=""></li>
        </ul>
    </div>
    <span id="next"></span>
</div>

<script type="text/javascript">
    window.onload = function () {
        // 获取元素
        var div = document.getElementById("box");
        var ul = document.getElementsByTagName("ul")[0];
        var li = document.getElementsByTagName("li");
        var back = document.getElementById("back");
        var next = document.getElementById("next");
        // 定义移动速度 正为向右走,负为向左走
        var speed = -2;
        // 复制一个ul ,让轮播衔接上
        ul.innerHTML = ul.innerHTML + ul.innerHTML;
        // 定义ul的长度是所有li长度和 （li[0]是第一个li,随便一个就可以,offsetWidth：是自身宽度）
        ul.style.width = li[0].offsetWidth * li.length + "px";

        function move() {
            // 向左移动ul , speed的距离上面定义-2
            ul.style.left = ul.offsetLeft + speed + "px";
            // 判断如果距#box左边的距离小于ul距离的一半,也就是一组图片的宽度时候(其实就是将轮播图拉回的过程)
            if (ul.offsetLeft < -ul.offsetWidth / 2) {
                // 重置距#box左边距离
                ul.style.left = "0";
            }
            // 这里是判断向右移动的拉回效果,原理如同上面
            if (ul.offsetLeft > 0) {

                ul.style.left = -ul.offsetWidth / 2 + "px";
            }
        }

        将移动方法放进计时器中
        var jsq = setInterval(move, 30);
        // 点击两个按钮设置speed的正负也就是移动的方向
        back.onclick = function () {
            speed = -2;
        }
        next.onclick = function () {
            speed = 2;
        }
        // 鼠标经过停止计时器
        ul.onmouseover = function () {
            clearInterval(jsq);
        }
        // 鼠标移开开启计时器
        ul.onmouseout = function () {
            jsq = setInterval(move, 30);
        }
    }
</script>
</body>
</html>

		var countrieArr = ['中国','美国'];
		var provinceArr = [['内蒙古','北京','上海'],['洛杉矶','华盛顿']];

		var countrieObj = document.getElementById('countrie');
		var provinceObj = document.getElementById('province');

		var countrieOpt = '';
		for(var i=0;i<countrieArr.length;i++)
			/*
				var begin = '<option value="'; var end = '">'; 
				var i = 1; 
				console.log(begin+i+end);
				VM451:1 <option value="1">

				var i=1;console.log('<option value="'+i+'">');
				VM491:1 <option value="1">
			*/
			countrieOpt += '<option value="'+i+'">'+countrieArr[i]+'</option>';
		}
		/*
			i = 0;
			i是否小于2  true
			执行代码
			i++
		
		
			i = 1;
			i是否小于2 true
			执行代码
			i++

			i是否小于2 false

		*/
		
		//countrieObj.innerHTML = countrieObj.innerHTML + countrieOpt;
		countrieObj.innerHTML +=  countrieOpt;

		//onchange表示内容发生改变，如果内容没有改变是不会触发的.
		countrieObj.onchange = function(){
		
			//获取当前国家下面的省份
			//var a = document.getElementById('countrie').value;
			//console.log(provinceArr[a]);
			
			//console.log(provinceArr[this.value]);
		
			var provinceOpt = '<option value="">|--请选择省份</option>';
		
			// this.value 表示获取当前对象的value值
			if(this.value != ''){
				for(var i=0;i<provinceArr[this.value].length;i++){
					provinceOpt += '<option>'+provinceArr[this.value][i]+'</option>';

			}
